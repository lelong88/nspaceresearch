```sql
-- Find all 4 series in this collection
SELECT * FROM curriculum_series WHERE curriculum_collection_id = '218zvh00';

-- Find all curriculums by series membership
SELECT c.id, c.content->>'title' as title
FROM curriculum c
JOIN curriculum_series_curriculum csc ON c.id = csc.curriculum_id
WHERE csc.curriculum_series_id IN ('tz30u1rs', '2aaul2m5', 'xiaaxk2i', 'z9k03u76');

-- Find all 20 curriculums by ID
SELECT id, content->>'title' as title, language, user_language, display_order, price
```
