## SQL Queries

```sql
-- Collection
SELECT * FROM curriculum_collections WHERE id = 'oqu57a8u';

-- All series in collection
SELECT cs.id, cs.title, cs.display_order
FROM curriculum_series cs
JOIN curriculum_collection_series ccs ON cs.id = ccs.curriculum_series_id
WHERE ccs.curriculum_collection_id = 'oqu57a8u'
ORDER BY cs.display_order;

-- All curriculums by series
SELECT c.id, c.content->>'title' as title, c.display_order, cs.title as series
FROM curriculum c
JOIN curriculum_series_curriculums csc ON c.id = csc.curriculum_id
JOIN curriculum_series cs ON csc.curriculum_series_id = cs.id
JOIN curriculum_collection_series ccs ON cs.id = ccs.curriculum_series_id
WHERE ccs.curriculum_collection_id = 'oqu57a8u'
  AND c.uid NOT LIKE '%_deleted'
ORDER BY cs.display_order, c.display_order;

-- Verify structure (session count per curriculum)
```
