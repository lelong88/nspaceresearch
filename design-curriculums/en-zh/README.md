   ```bash
   for f in en-zh/create_beg_*.py; do python "$f"; done
   for f in en-zh/create_preint_*.py; do python "$f"; done
   for f in en-zh/create_inter_*.py; do python "$f"; done
   ```

4. Run the SQL verification queries above to confirm counts, homogeneity, and pricing.
